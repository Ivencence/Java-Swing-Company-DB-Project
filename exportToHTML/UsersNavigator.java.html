public class UsersNavigator extends JFrame {

    private JTextField txtId;
    private JTextField txtUsername;
    private JTextField txtPassword;

    private JButton btnPrevious;
    private JButton btnNext;

    private Connection conn;

    private List<List<String>> data = new ArrayList<>();

    private int currentIndex = 0;

    public UsersNavigator() {
        connectDB();
        loadData();
        createUI();
        showRecord();
    }

    private void connectDB() {
        try {
            conn = DriverManager.getConnection(
                    "jdbc:mysql://localhost/classicmodels",
                    "root",
                    "My$ql_$erveR@2024"
            );
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Database connection failed");
        }
    }

    private void loadData() {
        try {
            Statement stmt = conn.createStatement();
            ResultSet rs = stmt.executeQuery("SELECT * FROM Users");

            while (rs.next()) {

                List<String> row = new ArrayList<>();

                row.add(String.valueOf(rs.getInt("id")));
                row.add(rs.getString("username"));
                row.add(rs.getString("password"));

                data.add(row);
            }

            rs.close();
            stmt.close();

        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Error loading data");
        }
    }

    private void createUI() {

        setTitle("Users Navigator");
        setSize(350, 200);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setLocationRelativeTo(null);

        JPanel panel = new JPanel(new GridLayout(4, 2, 5, 5));

        panel.add(new JLabel("ID:"));
        txtId = new JTextField();
        panel.add(txtId);

        panel.add(new JLabel("Username:"));
        txtUsername = new JTextField();
        panel.add(txtUsername);

        panel.add(new JLabel("Password:"));
        txtPassword = new JTextField();
        panel.add(txtPassword);

        btnPrevious = new JButton("Previous");
        btnNext = new JButton("Next");

        panel.add(btnPrevious);
        panel.add(btnNext);

        add(panel);

        btnPrevious.addActionListener(this::previousRecord);
        btnNext.addActionListener(this::nextRecord);
    }

    private void showRecord() {
        if (data.isEmpty()) return;

        List<String> row = data.get(currentIndex);

        txtId.setText(row.get(0));
        txtUsername.setText(row.get(1));
        txtPassword.setText(row.get(2));
    }

    private void previousRecord(ActionEvent e) {
        if (currentIndex > 0) {
            currentIndex--;
            showRecord();
        }
    }

    private void nextRecord(ActionEvent e) {
        if (currentIndex < data.size() - 1) {
            currentIndex++;
            showRecord();
        }
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(() -> {
            new UsersNavigator().setVisible(true);
        });
    }
}
